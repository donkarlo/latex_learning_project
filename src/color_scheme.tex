% ---- Donut wheel macro (no tables) ----
% #1 title
% #2 list entries: ColorName/TextColor/Hex
\newcommand{\DonutWheel}[2]{
    \begin{minipage}[t]{0.47\linewidth}
    \centering
    {\Large\bfseries #1\par}
    \vspace{0.7em}
    
    \begin{tikzpicture}
    \def\Rout{2.20}
    \def\Rin{0.65}
    \def\Rlab{1.50}
    \def\startAngle{90}
    \pgfmathsetmacro{\step}{360/8}
    
    \foreach \col/\tcol/\hex [count=\i from 0] in {#2} {
        \pgfmathsetmacro{\a}{\startAngle - \i*\step}
        \pgfmathsetmacro{\b}{\startAngle - (\i+1)*\step}
        \pgfmathsetmacro{\m}{(\a+\b)/2}
        
        \path[fill=\col, draw=white, line width=0.6pt]
        (\a:\Rout) arc (\a:\b:\Rout) --
        (\b:\Rin) arc (\b:\a:\Rin) -- cycle;
        
        % smaller hex text (copyable)
        \node[font=\ttfamily\fontsize{6}{7}\selectfont\bfseries, text=\tcol] at (\m:\Rlab) {\hex};
    }
    \fill[white] (0,0) circle (\Rin);
    \end{tikzpicture}
    \end{minipage}
}
